Scrolling Super Mario Clone body { margin: 0; overflow: hidden; } canvas { background: #5c94fc; display: block; margin: 0 auto; } const canvas = document.getElementById("gameCanvas"); const ctx = canvas.getContext("2d");

const marioImage = new Image(); marioImage.src = "https://i.imgur.com/OdL0XPt.png"; // Mario sprite

const tileSize = 50; const groundY = 350; let mario = { x: 100, y: groundY - tileSize, width: tileSize, height: tileSize, vy: 0, jumping: false }; let cameraX = 0; let keys = {};

document.addEventListener("keydown", e => keys[e.key] = true); document.addEventListener("keyup", e => keys[e.key] = false);

// Generate random map (20 screens wide) const mapWidth = 1600; const map = new Array(Math.floor(mapWidth / tileSize)).fill(0).map((_, i) => { return { x: i * tileSize, y: (Math.random() > 0.8) ? groundY - tileSize : groundY, height: (Math.random() > 0.8) ? tileSize : 50 }; });

function drawGround() { ctx.fillStyle = "#228B22"; for (let tile of map) { ctx.fillRect(tile.x - cameraX, tile.y, tileSize, tile.height); } }

function drawMario() { ctx.drawImage(marioImage, mario.x - cameraX, mario.y, mario.width, mario.height); }

function update() { if (keys['ArrowRight']) mario.x += 3; if (keys['ArrowLeft']) mario.x -= 3;

if (keys[' '] && !mario.jumping) { mario.vy = -10; mario.jumping = true; } mario.y += mario.vy; mario.vy += 0.5; // Simple collision with ground tiles for (let tile of map) { if ( mario.x + mario.width > tile.x && mario.x < tile.x + tileSize && mario.y + mario.height > tile.y && mario.y + mario.height < tile.y + tile.height + 10 ) { mario.y = tile.y - mario.height; mario.vy = 0; mario.jumping = false; } } // Camera follows Mario cameraX = mario.x - 150; if (cameraX < 0) cameraX = 0;

}

function loop() { ctx.clearRect(0, 0, canvas.width, canvas.height); drawGround(); drawMario(); update(); requestAnimationFrame(loop); }

marioImage.onload = loop;
